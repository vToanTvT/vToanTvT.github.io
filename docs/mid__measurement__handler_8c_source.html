78void meas_init_value_(S_Meas_TypeDef *pMeas, q15_t offset, uq31_t scale_factor,
79 float alpha, uq15_t high_limit_soft, float gain_f32,

132void meas_convert_to_q15_(S_Meas_TypeDef *pMeas)
133{
134 if(pMeas == NULL)
135 return;
136
137 if (pMeas->calc_method == CALC_METHOD_Q15) {
138 pMeas->value_q15 = _q15_mul(pMeas->raw >> 1, pMeas->scale_factor);
139 } else {
140 pMeas->value_q15 = _q15_mul(pMeas->raw >> 1, _q15(1));
141 }
142}

160void meas_convert_to_f32_(S_Meas_TypeDef *pMeas)
161{
162 if(pMeas == NULL)
163 return;
164
165 if (pMeas->calc_method == CALC_METHOD_Q15) {
166 pMeas->value_f32 = _q15_i2f((pMeas->value_q15) * pMeas->high_limit_soft);
167 } else {
168 pMeas->value_f32 = pMeas->raw * pMeas->gain_f32 + pMeas->offset_f32;
169 }
170}
