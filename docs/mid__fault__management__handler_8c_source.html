42_Bool fault_debounce_(_Bool condition, uint16_t timeout, uint16_t *counter)
43{
44 if (condition) {
45 if (*counter > timeout) {
46 *counter = 0;
47 return true;
48 } else {
49 (*counter)++;
50 return false;
51 }
52 } else {
53 *counter = 0;
54 return false;
55 }
56}

66void fault_update_status_(S_Fault_TypeDef *pFaultManage,
67 S_Meas_TypeDef *pVpsu,
68 S_Meas_TypeDef *pIpsu,
69 S_Meas_TypeDef *pVac,
70 S_Meas_TypeDef *pTpfc,
71 S_Meas_TypeDef *pTllc)
72{
73 if (fault_debounce_(FAULT_IS_LLC_OVP(pVpsu), FAULT_OVP_LLC_DEBOUNCE_TIME, &counter_ovp_llc))
74 temp_fault_status |= FAULT_STT_0;
75
76 if (fault_debounce_(FAULT_IS_LLC_OCP(pIpsu), FAULT_OCP_DEBOUNCE_TIME, &counter_ocp_llc))
77 temp_fault_status |= FAULT_STT_2;
78
79 if (FAULT_IS_LLC_SCP(pIpsu))
80 temp_fault_status |= FAULT_STT_3;
81
82 if (fault_debounce_(FAULT_IS_LLC_OTP(pTllc), FAULT_OTP_DEBOUNCE_TIME, &counter_otp_llc))
83 temp_fault_status |= FAULT_STT_4;
84
85 if (fault_debounce_(FAULT_IS_PFC_OTP(pTpfc), FAULT_OTP_DEBOUNCE_TIME, &counter_otp_pfc))
86 temp_fault_status |= FAULT_STT_5;
87
88 if (fault_debounce_(FAULT_IS_PFC_OVP(pVac), FAULT_OVP_PFC_DEBOUNCE_TIME, &counter_ovp_pfc))
89 temp_fault_status |= FAULT_STT_6;
90
91 if (fault_debounce_(FAULT_IS_PFC_UVP(pVac), FAULT_UVP_PFC_DEBOUNCE_TIME, &counter_uvp_pfc))
92 temp_fault_status |= FAULT_STT_7;
93
94 if (g_thisComp_.state == 1)
95 temp_fault_status |= FAULT_STT_12;
96
97 pFaultManage->fault_status = temp_fault_status;
98}
