173void llc_soft_start_()
174{
175 if(g_pwm_llc_config_.deadtime != SOFT_START_DEADTIME_TARGET){
176 if (s_cnt_delay_softstart++ >= SOFT_START_RATE_DEA) {
177 s_cnt_delay_softstart = 0;
178 RampDown(SOFT_START_DEADTIME_TARGET, SOFT_START_DEADTIME_STEP, &g_pwm_llc_config_.deadtime);
179 pwm_set_deadtime_(&g_pwm_llc_config_, g_pwm_llc_config_.deadtime);
180 }
181 } else {
182 if (s_cnt_delay_softstart++ >= SOFT_START_RATE_FRE) {
183 s_cnt_delay_softstart = 0;
184 RampDown(SOFT_START_FREQ_TARGET,SOFT_START_FREQ_STEP,&g_pwm_llc_config_.frequency);
185 pwm_set_frequency_(&g_pwm_llc_config_, g_pwm_llc_config_.frequency);
186 }
187 }
188
189 int temp_duty = ((int)1000000000/(int)g_pwm_llc_config_.frequency -
190 2*(int)g_pwm_llc_config_.deadtime)/ 2* (int)DUTY_MAX /
191 (1000000000/(int)g_pwm_llc_config_.frequency) ;
192
193 if(temp_duty < 0){
194 g_pwm_llc_config_.duty = 0;
195 } else {
196 g_pwm_llc_config_.duty = temp_duty;
197 }
198 pwm_set_duty_(&g_pwm_llc_config_, g_pwm_llc_config_.duty);
199 pwm_enable_output_(&g_pwm_llc_config_);
200}

284void llc_regulator_hybrid_loop()
285{
286 g_LLC_Vol_Controller.feedback_val = g_bat_voltage_inst.value_q15;
287 g_LLC_Dut_Controller.feedback_val = g_bat_current_inst.value_q15;
288 g_LLC_Cur_Controller.feedback_val = g_bat_current_inst.value_q15;
289
290 if ( g_LLC_Vol_Controller.feedback_val > VOLTAGE_THRESHOLD_Q15 ) {
291 g_LLC_Vol_Controller.high_limit = CURRENT_THRESHOLD_HIGH_Q15;
292 } else {
293 g_LLC_Vol_Controller.high_limit = CURRENT_THRESHOLD_LOW_Q15;
294 }
295
296 controller_process_pi_(&g_LLC_Vol_Controller, 1);
297 controller_anti_windup_(&g_LLC_Vol_Controller);
298
299 RampUpRate(&g_LLC_Dut_Controller.ref_val, LOW_LIMIT_VOL_Q15, g_LLC_Vol_Controller.output, RATE_RAMP_UP_1_x_dt_Q15);
300 RampUpRate(&g_LLC_Cur_Controller.ref_val, DUT_REF_CUR_Q15 , g_LLC_Vol_Controller.output, RATE_RAMP_UP_2_x_dt_Q15);
301
302 if ( g_LLC_Config.mode == LLC_MODE_PFM ) {
303 if ( (g_LLC_Cur_Controller.output >= g_LLC_Cur_Controller.high_limit) &&
304 (g_LLC_Cur_Controller.feedback_val > g_LLC_Cur_Controller.ref_val) ) {
305
306 g_LLC_Config.mode = LLC_MODE_PWM;
307 controller_process_pi_(&g_LLC_Dut_Controller, 1);
308 controller_anti_windup_(&g_LLC_Dut_Controller);
309
310 g_pwm_llc_config_.duty = (uint32_t)((_q15_i2f(g_LLC_Dut_Controller.output)) * DUTY_MAX);
311 pwm_set_duty_(&g_pwm_llc_config_, g_pwm_llc_config_.duty);
312 pwm_set_deadtime_(&g_pwm_llc_config_, LLC_END_DEADTIME);
313 return;
314 }
315 else {
316 controller_process_pi_(&g_LLC_Cur_Controller, 1);
317 controller_anti_windup_(&g_LLC_Cur_Controller);
318
319 g_pwm_llc_config_.frequency = (uint32_t)((_q15_i2f(g_LLC_Cur_Controller.output)) * FREQ_MAX);
320 pwm_set_frequency_(&g_pwm_llc_config_, g_pwm_llc_config_.frequency);
321 pwm_set_deadtime_(&g_pwm_llc_config_, LLC_END_DEADTIME);
322 return;
323 }
324 }
325 else if ( g_LLC_Config.mode == LLC_MODE_PWM ) {
326 if ( (g_LLC_Dut_Controller.output >= g_LLC_Dut_Controller.high_limit) &&
327 (g_LLC_Dut_Controller.feedback_val < g_LLC_Dut_Controller.ref_val) ) {
328
329 g_LLC_Config.mode = LLC_MODE_PFM;
330 controller_process_pi_(&g_LLC_Cur_Controller, 1);
331 controller_anti_windup_(&g_LLC_Cur_Controller);
332
333 g_pwm_llc_config_.frequency = (uint32_t)((_q15_i2f(g_LLC_Cur_Controller.output)) * FREQ_MAX);
334 pwm_set_frequency_(&g_pwm_llc_config_, g_pwm_llc_config_.frequency);
335 pwm_set_deadtime_(&g_pwm_llc_config_, LLC_END_DEADTIME);
336 return;
337 }

342 g_pwm_llc_config_.duty = (uint32_t)((_q15_i2f(g_LLC_Dut_Controller.output)) * DUTY_MAX);
