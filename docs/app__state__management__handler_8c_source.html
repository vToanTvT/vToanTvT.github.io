165void state_update(S_EVSE_t *pEVSE){
170 switch (pEVSE->State) {
171 case EVSE_STATE_IDLE:
172 /* Processing: state processing */
173 state_idleHandle();
174
175 /* Post-processing: transit to next state */
176 if (g_fault_instance_.fault_mode != FAULT_NONE) {
177 pEVSE->State = EVSE_STATE_FAULT;
178 }
179 else if ( state_isDebounceFinished(STATE_TRANSITION_IDLE_TO_SOFTSTART, STATE_IDLE_DEBOUNCE_US, &pEVSE->DebounceCounter.state_idle_counter) ) {
180 pEVSE->State = EVSE_STATE_SOFTSTART;
181 }
182 break;
183
184 case EVSE_STATE_SOFTSTART:
185 /* Processing: state processing */
186 state_softstartHandle();
187
188 /* Post-processing: transit to next state */
189 if (g_fault_instance_.fault_mode != FAULT_NONE) {
190 pEVSE->State = EVSE_STATE_FAULT;
191 }
192 else if ( state_isDebounceFinished(STATE_TRANSITION_SOFTSTART_TO_HANDSHAKE, STATE_SOFTSTART_DEBOUNCE_US, &pEVSE->DebounceCounter.state_softstart_counter) ) {
193 pEVSE->State = EVSE_STATE_HANDSHAKE;
194 pwm_disable_output_(&g_pwm_llc_config_);
195 }
196 break;
197
198 case EVSE_STATE_HANDSHAKE:
199 /* Processing: state processing */
200 state_handshakeHandle();
201
202 /* Post-processing: transit to next state */
203 if (g_fault_instance_.fault_mode != FAULT_NONE) {
204 pEVSE->State = EVSE_STATE_FAULT;
205 }
206 else if ( state_isDelayFinished(&pEVSE->DelayCounter.state_handshake_counter, STATE_HANDSHAKE_DELAY_US, SAMPLING_TIME_US) ) {
207 pEVSE->State = EVSE_STATE_CHARGING;
208 }
209 break;
210
211 case EVSE_STATE_CHARGING:
212 /* Processing: state processing */
213 pwm_enable_output_(&g_pwm_llc_config_);
214 state_chargingHandle();
215
216 /* Post-processing: transit to next state */
217 if (g_fault_instance_.fault_mode != FAULT_NONE) {
218 pEVSE->State = EVSE_STATE_FAULT;
219 }
220 else if ( state_isDebounceFinished(STATE_TRANSITION_CHARGING_TO_FULLYCHARGED, STATE_CHARGING_DEBOUNCE_US, &pEVSE->DebounceCounter.state_charging_counter) ) {
221 pEVSE->State = EVSE_STATE_FULLYCHARGED;
222 }
223 break;
224
225 case EVSE_STATE_FULLYCHARGED:
226 /* Processing: state processing */
227 state_fullychargedHandle();
228
229 /* Post-processing: transit to next state */
230 if (g_fault_instance_.fault_mode != FAULT_NONE) {
231 pEVSE->State = EVSE_STATE_FAULT;
232 }
233 else if (state_isDebounceFinished(STATE_TRANSITION_FULLYCHARGED_TO_IDLE, STATE_FULLYCHARGED_DEBOUNCE_US, &pEVSE->DebounceCounter.state_fullycharged_counter)){
234 pEVSE->State = EVSE_STATE_IDLE;
235 }
236 break;
237
238 case EVSE_STATE_FAULT:
239 /* Processing: state processing */
240 state_faultHandle();
241
242 /* Post-processing: transit to next state */
243 if ((g_fault_instance_.fault_mode == FAULT_LATCH)) {
244 pEVSE->State = EVSE_STATE_SHUTDOWN;
245 }
246 else if(state_isDelayFinished(&pEVSE->DelayCounter.state_fault_counter, STATE_FAULT_DELAY_US, SAMPLING_TIME_US))
247 {
248 /* Transit to IDLE State and erase fault flags */
249 g_fault_instance_.fault_save[pEVSE->faultCounter] = g_fault_instance_.fault_status;
250 if(!((g_fault_instance_.fault_status == 1 ) &&(g_bat_current_inst.value_q15 <= FAULT_RELEASED_LLC_OCP_Q15))){
251 pEVSE->faultCounter++;
252 }
253 pEVSE->State = EVSE_STATE_AUTORECOVERY;
254 }
255 break;
256
257 case EVSE_STATE_AUTORECOVERY:
258 /* Processing: state processing */
259 state_autorecoveryHandle();
260
261 /* Post-processing: transit to next state */
262 if ( (pEVSE->faultCounter >= pEVSE->faultLimit) ) {
263 pEVSE->State = EVSE_STATE_SHUTDOWN;
264 }
265 else if (state_isDebounceFinished( fault_release_(&g_fault_instance_, &g_dc_voltage_inst, &g_bat_current_inst,
266 &g_ac_voltage_inst, &g_thermal_pfc_inst, &g_thermal_llc_inst), STATE_AUTORECOVERY_DEBOUNCE_US, &pEVSE->DebounceCounter.state_autorecovery_counter)){
267 fault_standby_();
268 pEVSE->State = EVSE_STATE_IDLE;
269 }
270 break;
271
272 case EVSE_STATE_SHUTDOWN:
273 /* Processing: state processing */
274 state_shutdownHandle();
275 /* Post-processing: transit to next state */
276 break;
277
278 default:
279 break;
280 }
281 return;
282}

380bool state_isDelayFinished(uint32_t *p_counter, const uint32_t delay_ms, const uint32_t sampling_time_ms)
381{
382 if (p_counter == NULL || sampling_time_ms == 0) {
383 return false;
384 }
385
386 const uint32_t target_count = (delay_ms / sampling_time_ms);
387
388 if (*p_counter < target_count) {
389 (*p_counter)++;
390 return false;
391 }
392
393 *p_counter = 0;
394 return true;
395}

404bool state_isDebounceFinished(bool condition, uint32_t timeout, uint32_t *p_counter)
405{
406 if (p_counter == NULL) {
407 return false;
408 }
409
410 if (condition) {
411 if (*p_counter > timeout) {
412 *p_counter = 0;
413 return true;
414 }
415 (*p_counter)++;
416 return false;
417 }
418
419 *p_counter = 0;
420 return false;
421}

432static void state_error_handler_(uint8_t error_code)
433{
434 s_state_control.error_code = error_code;
435 s_state_control.retry_count++;
436
437 if (s_state_control.retry_count >= K_STATE_MAX_RETRIES_) {
438 thisEVSE.State = EVSE_STATE_FAULT;

447static void state_update_indicators_(E_EVSE_State_t current_state)
448{
449 /* Update LED indicators based on state */
450 switch (current_state) {
451 case EVSE_STATE_IDLE:
452 /* Green LED slow blink */
453 break;
454 case EVSE_STATE_CHARGING:
455 /* Green LED solid */
456 break;
457 case EVSE_STATE_FAULT:
458 /* Red LED fast blink */
459 break;
460 default:
461 /* All LEDs off */
462 break;
463 }
464}

472static bool state_validate_transition_(E_EVSE_State_t from_state, E_EVSE_State_t to_state)
473{
474 /* Check for invalid states */
475 if (!M_STATE_CHECK_VALID_(from_state) || !M_STATE_CHECK_VALID_(to_state)) {
476 return false;
477 }
478
479 /* Validate specific transitions */
480 switch (from_state) {
481 case EVSE_STATE_IDLE:
482 return (to_state == EVSE_STATE_SOFTSTART || to_state == EVSE_STATE_FAULT);
483
484 case EVSE_STATE_CHARGING:
485 return (to_state == EVSE_STATE_FULLYCHARGED || to_state == EVSE_STATE_FAULT);
486
487 case EVSE_STATE_FAULT:
488 return (to_state == EVSE_STATE_AUTORECOVERY || to_state == EVSE_STATE_SHUTDOWN);
489
490 default:
491 return false;
492 }
493}
