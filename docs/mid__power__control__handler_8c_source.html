159 uq15_t pwrctrl_get_median_(uq15_t* values) {
160 int i;
161 int alt_values[4];
162 for (i = 0; i < 4; i++) {
163 alt_values[i] = values[i];
164 }
165
166 if (alt_values[0] > alt_values[1]) pwrctrl_value_swap_(&alt_values[0], &alt_values[1]);
167 if (alt_values[2] > alt_values[3]) pwrctrl_value_swap_(&alt_values[2], &alt_values[3]);
168
169 if (alt_values[0] > alt_values[2]) pwrctrl_value_swap_(&alt_values[0], &alt_values[2]);
170 if (alt_values[1] > alt_values[3]) pwrctrl_value_swap_(&alt_values[1], &alt_values[3]);
171 if (alt_values[1] > alt_values[2]) pwrctrl_value_swap_(&alt_values[1], &alt_values[2]);
172
173 return _q15_add(alt_values[1] , alt_values[2]) / 2;
174 }

218 void pwrctrl_compute_temp_coef_(S_PowerCtrl_t *pPowerControl) {
219 // Find the maximum temperature among thermal_PFC_inst, thermal_LLC_inst, thermal_psu_inst
220 int max_temp = g_thermal_pfc_inst.value_q15;
221 if (g_thermal_llc_inst.value_q15 > max_temp) max_temp = g_thermal_llc_inst.value_q15;
222
223 pwrctrl_update_buffer_(pPowerControl->temperature, TEMP_CNT, max_temp);
224 pPowerControl->temp_avg = pwrctrl_calc_average_(pPowerControl->temperature, TEMP_CNT);
225 pPowerControl->temp_med = pwrctrl_get_median_(pPowerControl->temperature);
226
227 pPowerControl->temp_coef_q15 = pwrctrl_lookup_coef_q15_(pPowerControl->temp_med, s_temp_table, s_temp_table_size);
228 return;
229 }

258 void pwrctrl_update_coef_(S_PowerCtrl_t *pPowerControl) {
259 pwrctrl_compute_temp_coef_(pPowerControl);
260 pwrctrl_compute_voltage_coef_(pPowerControl);
261 // Find minimum coefficients
262 pPowerControl->derate_coef_q15 = pPowerControl->temp_coef_q15;
263 if (pPowerControl->derate_coef_q15 > pPowerControl->voltage_coef_q15){
264 pPowerControl->derate_coef_q15 = pPowerControl->voltage_coef_q15;
265 }
266
267 pPowerControl->derate_coef_f32 = _q15_i2f(pPowerControl->derate_coef_q15);
268 pPowerControl->voltage_coef_f32 = _q15_i2f(pPowerControl->voltage_coef_q15);
269 pPowerControl->temp_coef_f32 = _q15_i2f(pPowerControl->temp_coef_q15);
270 return;
271 }
