77void controller_process_pi_(S_PI_TypeDef *pController, uq15_t Status)
78{
79 if (pController == NULL)
80 return;
81
82 pController->state_control = Status;
83
84 if (Status == 0) { // Disabled
85 pController->error = 0;
86 pController->proportional = 0;
87 pController->integral = 0;
88 pController->output = 0;
89 return;
90 }
91
92 /* Calculate error */
93 pController->error = _q15_sub(pController->ref_val, pController->feedback_val);
94
95 /* Proportional term */
96 pController->proportional = _q15_mul(pController->error, pController->Kp);
97
98 /* Integral term */
99 pController->integral += _q15_mul(pController->error, pController->Ki_x_Tz);
100
101 /* Calculate output */
102 pController->output = _q15_add(pController->proportional, pController->integral);
103}

112void controller_anti_windup_(S_PI_TypeDef *pController)
113{
114 if (pController == NULL)
115 return;
116
117 if (pController->output > pController->high_limit) {
118 pController->output = pController->high_limit;
119 pController->integral = _q15_sub(pController->high_limit, pController->proportional);
120 }
121 else if (pController->output < pController->low_limit) {
122 pController->output = pController->low_limit;
123 pController->integral = _q15_sub(pController->low_limit, pController->proportional);
124 }
