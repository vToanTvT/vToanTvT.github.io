78void controller_process_pi_(S_PI_TypeDef *pController, uq15_t Status)
79{
80 pController->state_control = Status;
81 if (Status == /* DISABLE */ 0) {
82 pController->error = 0;
83 pController->proportional = 0;
84 pController->integral = 0;
85 pController->output = 0;
86 return;
87 }
88 else if (Status == /* ENABLE */ 1) {
89 /* Calculate error*/
90
91 pController->error = _q15_sub(pController->ref_val, pController->feedback_val);
92
93 /* Proportional term*/
94 pController->proportional = _q15_mul(pController->error, pController->Kp);
95
96 /* Integral term*/
97 pController->integral += _q15_mul(pController->error, pController->Ki_x_Tz);
98
99 /* Calculate output*/
100 pController->output = _q15_add(pController->proportional, pController->integral);
101 }
102 return;
103}
