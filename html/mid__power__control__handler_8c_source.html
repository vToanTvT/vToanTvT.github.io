157 uq15_t pwrctrl_get_median_(uq15_t* values) {
158 int i;
159 int alt_values[4];
160 for (i = 0; i < 4; i++) {
161 alt_values[i] = values[i];
162 }
163
164 if (alt_values[0] > alt_values[1]) pwrctrl_value_swap_(&alt_values[0], &alt_values[1]);
165 if (alt_values[2] > alt_values[3]) pwrctrl_value_swap_(&alt_values[2], &alt_values[3]);
166
167 if (alt_values[0] > alt_values[2]) pwrctrl_value_swap_(&alt_values[0], &alt_values[2]);
168 if (alt_values[1] > alt_values[3]) pwrctrl_value_swap_(&alt_values[1], &alt_values[3]);
169 if (alt_values[1] > alt_values[2]) pwrctrl_value_swap_(&alt_values[1], &alt_values[2]);
170
171 return _q15_add(alt_values[1] , alt_values[2]) / 2;
172 }

216 void pwrctrl_compute_temp_coef_(S_PowerCtrl_t *pPowerControl) {
217 // Find the maximum temperature among thermal_PFC_inst, thermal_LLC_inst, thermal_psu_inst
218 int max_temp = g_thermal_pfc_inst.value_q15;
219 if (g_thermal_llc_inst.value_q15 > max_temp) max_temp = g_thermal_llc_inst.value_q15;
220
221 pwrctrl_update_buffer_(pPowerControl->temperature, TEMP_CNT, max_temp);
222 pPowerControl->temp_avg = pwrctrl_calc_average_(pPowerControl->temperature, TEMP_CNT);
223 pPowerControl->temp_med = pwrctrl_get_median_(pPowerControl->temperature);
224
225 pPowerControl->temp_coef_q15 = pwrctrl_lookup_coef_q15_(pPowerControl->temp_med, s_temp_table, s_temp_table_size);
226 return;
227 }

256 void pwrctrl_update_coef_(S_PowerCtrl_t *pPowerControl) {
257 pwrctrl_compute_temp_coef_(pPowerControl);
258 pwrctrl_compute_voltage_coef_(pPowerControl);
259 // Find minimum coefficients
260 pPowerControl->derate_coef_q15 = pPowerControl->temp_coef_q15;
261 if (pPowerControl->derate_coef_q15 > pPowerControl->voltage_coef_q15){
262 pPowerControl->derate_coef_q15 = pPowerControl->voltage_coef_q15;
263 }
264
265 pPowerControl->derate_coef_f32 = _q15_i2f(pPowerControl->derate_coef_q15);
266 pPowerControl->voltage_coef_f32 = _q15_i2f(pPowerControl->voltage_coef_q15);
267 pPowerControl->temp_coef_f32 = _q15_i2f(pPowerControl->temp_coef_q15);
268 return;
