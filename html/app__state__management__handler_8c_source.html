162void state_update(S_EVSE_t *pEVSE){
167 switch (pEVSE->State) {
168 case EVSE_STATE_IDLE:
169 /* Processing: state processing */
170 state_idleHandle();
171
172 /* Post-processing: transit to next state */
173 if (g_fault_instance_.fault_mode != FAULT_NONE) {
174 pEVSE->State = EVSE_STATE_FAULT;
175 }
176 else if ( state_isDebounceFinished(STATE_TRANSITION_IDLE_TO_SOFTSTART, STATE_IDLE_DEBOUNCE_US, &pEVSE->DebounceCounter.state_idle_counter) ) {
177 pEVSE->State = EVSE_STATE_SOFTSTART;
178 }
179 break;
180
181 case EVSE_STATE_SOFTSTART:
182 /* Processing: state processing */
183 state_softstartHandle();
184
185 /* Post-processing: transit to next state */
186 if (g_fault_instance_.fault_mode != FAULT_NONE) {
187 pEVSE->State = EVSE_STATE_FAULT;
188 }
189 else if ( state_isDebounceFinished(STATE_TRANSITION_SOFTSTART_TO_HANDSHAKE, STATE_SOFTSTART_DEBOUNCE_US, &pEVSE->DebounceCounter.state_softstart_counter) ) {
190 pEVSE->State = EVSE_STATE_HANDSHAKE;
191 pwm_disable_output_(&g_pwm_llc_config_);
192 }
193 break;
194
195 case EVSE_STATE_HANDSHAKE:
196 /* Processing: state processing */
197 state_handshakeHandle();
198
199 /* Post-processing: transit to next state */
200 if (g_fault_instance_.fault_mode != FAULT_NONE) {
201 pEVSE->State = EVSE_STATE_FAULT;
202 }
203 else if ( state_isDelayFinished(&pEVSE->DelayCounter.state_handshake_counter, STATE_HANDSHAKE_DELAY_US, SAMPLING_TIME_US) ) {
204 pEVSE->State = EVSE_STATE_CHARGING;
205 }
206 break;
207
208 case EVSE_STATE_CHARGING:
209 /* Processing: state processing */
210 pwm_enable_output_(&g_pwm_llc_config_);
211 state_chargingHandle();
212
213 /* Post-processing: transit to next state */
214 if (g_fault_instance_.fault_mode != FAULT_NONE) {
215 pEVSE->State = EVSE_STATE_FAULT;
216 }
217 else if ( state_isDebounceFinished(STATE_TRANSITION_CHARGING_TO_FULLYCHARGED, STATE_CHARGING_DEBOUNCE_US, &pEVSE->DebounceCounter.state_charging_counter) ) {
218 pEVSE->State = EVSE_STATE_FULLYCHARGED;
219 }
220 break;
221
222 case EVSE_STATE_FULLYCHARGED:
223 /* Processing: state processing */
224 state_fullychargedHandle();
225
226 /* Post-processing: transit to next state */
227 if (g_fault_instance_.fault_mode != FAULT_NONE) {
228 pEVSE->State = EVSE_STATE_FAULT;
229 }
230 else if (state_isDebounceFinished(STATE_TRANSITION_FULLYCHARGED_TO_IDLE, STATE_FULLYCHARGED_DEBOUNCE_US, &pEVSE->DebounceCounter.state_fullycharged_counter)){
231 pEVSE->State = EVSE_STATE_IDLE;
232 }
233 break;
234
235 case EVSE_STATE_FAULT:
236 /* Processing: state processing */
237 state_faultHandle();
238
239 /* Post-processing: transit to next state */
240 if ((g_fault_instance_.fault_mode == FAULT_LATCH)) {
241 pEVSE->State = EVSE_STATE_SHUTDOWN;
242 }
243 else if(state_isDelayFinished(&pEVSE->DelayCounter.state_fault_counter, STATE_FAULT_DELAY_US, SAMPLING_TIME_US))
244 {
245 /* Transit to IDLE State and erase fault flags */
246 g_fault_instance_.fault_save[pEVSE->faultCounter] = g_fault_instance_.fault_status;
247 if(!((g_fault_instance_.fault_status == 1 ) &&(g_bat_current_inst.value_q15 <= FAULT_RELEASED_LLC_OCP_Q15))){
248 pEVSE->faultCounter++;
249 }
250 pEVSE->State = EVSE_STATE_AUTORECOVERY;
251 }
252 break;
253
254 case EVSE_STATE_AUTORECOVERY:
255 /* Processing: state processing */
256 state_autorecoveryHandle();
257
258 /* Post-processing: transit to next state */
259 if ( (pEVSE->faultCounter >= pEVSE->faultLimit) ) {
260 pEVSE->State = EVSE_STATE_SHUTDOWN;
261 }
262 else if (state_isDebounceFinished( fault_release_(&g_fault_instance_, &g_dc_voltage_inst, &g_bat_current_inst,
263 &g_ac_voltage_inst, &g_thermal_pfc_inst, &g_thermal_llc_inst), STATE_AUTORECOVERY_DEBOUNCE_US, &pEVSE->DebounceCounter.state_autorecovery_counter)){
264 fault_standby_();
265 pEVSE->State = EVSE_STATE_IDLE;
266 }
267 break;
268
269 case EVSE_STATE_SHUTDOWN:
270 /* Processing: state processing */
271 state_shutdownHandle();
272 /* Post-processing: transit to next state */
273 break;
274
275 default:
276 break;
277 }
278 return;
279}

377bool state_isDelayFinished(uint32_t *p_counter, const uint32_t delay_ms, const uint32_t sampling_time_ms)
378{
379 if (p_counter == NULL || sampling_time_ms == 0) {
380 return false;
381 }
382
383 const uint32_t target_count = (delay_ms / sampling_time_ms);
384
385 if (*p_counter < target_count) {
386 (*p_counter)++;
387 return false;
388 }
389
390 *p_counter = 0;
391 return true;
392}

401bool state_isDebounceFinished(bool condition, uint32_t timeout, uint32_t *p_counter)
402{
403 if (p_counter == NULL) {
404 return false;
405 }
406
407 if (condition) {
408 if (*p_counter > timeout) {
409 *p_counter = 0;
410 return true;
411 }
412 (*p_counter)++;
413 return false;
414 }
415
416 *p_counter = 0;
417 return false;
418}

445{

469static bool state_validate_transition_(E_EVSE_State_t from_state, E_EVSE_State_t to_state)
470{
471 /* Check for invalid states */
472 if (!M_STATE_CHECK_VALID_(from_state) || !M_STATE_CHECK_VALID_(to_state)) {
473 return false;
474 }
475
476 /* Validate specific transitions */
477 switch (from_state) {
478 case EVSE_STATE_IDLE:
479 return (to_state == EVSE_STATE_SOFTSTART || to_state == EVSE_STATE_FAULT);
480
481 case EVSE_STATE_CHARGING:
482 return (to_state == EVSE_STATE_FULLYCHARGED || to_state == EVSE_STATE_FAULT);
483
484 case EVSE_STATE_FAULT:
485 return (to_state == EVSE_STATE_AUTORECOVERY || to_state == EVSE_STATE_SHUTDOWN);
486
487 default:
488 return false;
489 }
