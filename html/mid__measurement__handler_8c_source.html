126void meas_convert_to_q15_(S_Meas_TypeDef *pMeas)
127{
128 if(pMeas == NULL)
129 return;
130
131 if (pMeas->calc_method == CALC_METHOD_Q15) {
132 pMeas->value_q15 = _q15_mul(pMeas->raw >> 1, pMeas->scale_factor);
133 } else {
134 pMeas->value_q15 = _q15_mul(pMeas->raw >> 1, _q15(1));
135 }
136}

154void meas_convert_to_f32_(S_Meas_TypeDef *pMeas)
155{
156 if(pMeas == NULL)
157 return;
158
159 if (pMeas->calc_method == CALC_METHOD_Q15) {
160 pMeas->value_f32 = _q15_i2f((pMeas->value_q15) * pMeas->high_limit_soft);
161 } else {
162 pMeas->value_f32 = pMeas->raw * pMeas->gain_f32 + pMeas->offset_f32;
163 }
164}
